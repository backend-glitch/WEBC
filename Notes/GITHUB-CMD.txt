🧠 All Git & GitHub Commands for VS Code

⚙️ Setup & Configuration
# Set your Git username and email (only once per system)
git config --global user.name "your-username"
git config --global user.email "your-email@example.com"

# Check configuration
git config --list

📂 Initialize a Repository
# Create a new Git repository in your current folder
git init

# Clone an existing repository from GitHub
git clone https://github.com/username/repository.git

📄 Staging & Committing
# Check the status of your files
git status

# Add a specific file
git add filename

# Add all files
git add .

# Commit your changes with a message
git commit -m "Add feature or fix bug"

# View commit history
git log

🔄 Branching
# Check current branch
git branch

# Create a new branch
git branch feature-branch

# Switch to another branch
git checkout feature-branch

# Create & switch to a new branch
git checkout -b new-feature

# Merge a branch into main
git merge feature-branch

☁️ Pushing & Pulling
# Add remote repository (link local repo to GitHub)
git remote add origin https://github.com/username/repository.git

# Push your changes to GitHub
git push -u origin main

# Pull latest changes from GitHub
git pull origin main

# View remote URLs
git remote -v

🧹 Undoing & Resetting
# Undo changes in a file (before staging)
git checkout -- filename

# Unstage a file (keep changes)
git reset filename

# Undo last commit (keep files)
git reset --soft HEAD~1

# Undo last commit completely (remove changes)
git reset --hard HEAD~1


⚡ Fork Workflow (For Contributing to Others’ Projects)
# 1️⃣ Fork the repository on GitHub (from web)

# 2️⃣ Clone your fork to local
git clone https://github.com/your-username/forked-repo.git

# 3️⃣ Move into the folder
cd forked-repo

# 4️⃣ Add the original repo as upstream
git remote add upstream https://github.com/original-owner/repo.git

# 5️⃣ Fetch latest changes from upstream
git fetch upstream

# 6️⃣ Merge upstream changes into your local main branch
git checkout main
git merge upstream/main

# 7️⃣ Push updates to your fork
git push origin main

🚀 Basic Git & GitHub Workflow
# Step 1: Initialize a local repository
git init

# Step 2: Link to your GitHub repo
git remote add origin https://github.com/username/repository.git

# Step 3: Stage changes
git add .

# Step 4: Commit with a message
git commit -m "Initial commit"

# Step 5: Push changes to GitHub
git push -u origin main

# Step 6: Pull updates if others changed the repo
git pull origin main


# To untrack a file
git rm --cached filename

# to check history
git log --oneline

# to check diff in two timelines

git diff --staged
